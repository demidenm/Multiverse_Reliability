#!/bin/bash

# define variables
curr_dir=`pwd`
ses=SESSION
task=MID
model=MODEL
# wilson-sub, wilson-supra
mask_label=MASKLABEL
# session or run
type=TYPE
mask_path=${curr_dir}/../brain_mask/MNI152_${mask_label}.nii.gz
inp_fold=INPFOLD
out_fold=/scratch.global/${USER}/analyses_reliability/icc_perm/${type}
script_dir=${curr_dir}/..

echo "#### Starting ICC estimation ####" 

# Define directories

[ ! -d ${out_fold} ] && echo "ICC folder out directory exists" | mkdir -p ${out_fold} 

# run python script
echo "sub: ${sub}"
echo "model: ${model}"
echo "Inp: ${inp_fold}"
echo "Out: ${out_fold}"

python ${script_dir}/compute_icc_permutations.py --sample abcd --ses ${ses} --type ${type} --model ${model} --mask ${mask_path} --mask_label ${mask_label} --inp_path ${inp_fold} --output ${out_fold}
icc_error=$?

if [ ${icc_error} -eq 0 ]; then
        echo "Python ICC permutation ${model} completed successfully!"
else
        echo "Python ICC permutation ${model} failed."
        exit 1
fi
